## ✅ الحل المقترح

### 1. تعديل `modules/app.js`
إضافة دعم serverless mode:

```javascript
class App {
  constructor(options = {}) {
    this.app = express();
    this.isServerless = options.isServerless || false;
    this.corsConfig = options.corsConfig || null;
    this.setupApp();
  }

  // دالة جديدة للحصول على التطبيق بدون تشغيل السيرفر
  getExpressApp() {
    return this.app;
  }

  setupMiddleware() {
    // CORS مخصص للـ serverless
    if (this.isServerless && this.corsConfig) {
      this.app.use(this.corsConfig);
    } else {
      this.app.use(cors(config.security.cors));
    }
    
    // ... باقي الكود
  }
}
```

### 2. تبسيط `vercel-server.js`
استخدام App class:

```javascript
const App = require('./modules/app');
const mongoose = require('mongoose');

// ... DB connection code

// CORS config for serverless
const corsMiddleware = (req, res, next) => {
  setCorsHeaders(req, res);
  if (req.method === 'OPTIONS') return res.status(204).end();
  next();
};

module.exports = async (req, res) => {
  setCorsHeaders(req, res);
  if (req.method === 'OPTIONS') return res.status(204).end();

  try {
    if (!MONGO_URI) {
      return res.status(500).json({ 
        success: false, 
        message: 'MONGO_URI is not set' 
      });
    }

    await connectDB().catch(err => {
      console.error('[Vercel] DB error:', err.message);
    });

    // استخدام App class
    const appInstance = new App({ 
      isServerless: true,
      corsConfig: corsMiddleware 
    });
    const expressApp = appInstance.getExpressApp();
    
    return expressApp(req, res);

  } catch (error) {
    console.error('[Vercel] Error:', error.message);
    if (!res.headersSent) {
      return res.status(500).json({ 
        success: false, 
        message: 'Server error' 
      });
    }
  }
};
```

## ⚠️ نقاط الحذر

### 1. CORS Configuration
- تأكد من أن CORS يعمل بشكل صحيح في كلا البيئتين
- اختبر مع domains مختلفة

### 2. Database Connection
- تأكد من أن connection pooling يعمل
- راقب memory leaks في serverless

### 3. Middleware Order
- ترتيب middleware مهم جداً
- CORS يجب أن يكون أول شيء

### 4. Error Handling
- تأكد من أن errors تُعالج بشكل صحيح
- لا تكشف معلومات حساسة
